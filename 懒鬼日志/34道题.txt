基于之上的基础在进行分组就好
select
e.deptno,e.ename,e.sal,s.grade,avg(s.grade) as avggrade
from
emp e
join 
salgrade s
on
e.sal between s.losal and s.hisal
group by
e.deptno
order by
s.grade;
